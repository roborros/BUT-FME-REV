## 1.program vypise fibonaciho posloupnost 

```c
	int i, f1=0, f2=1, dalsi;
	int n;
	
	n = 11;
		
	for(i=1; i<=n; i++){
		printf("%u\n", f1);
		dalsi = f1 + f2;
		f1 = f2;
		f2 = dalsi;
	}
```

	int n = 21;
		
	if((n%3) != (n%7)){
